<script>
    // رابط الجدول الخاص بك بصيغة CSV لسهولة القراءة
    const sheetID = '10RiRpOwdCtq8w_WwLJhUkUFxFPMkI595yiBqw2ET0pg';
    const url = `https://docs.google.com/spreadsheets/d/${sheetID}/gviz/tq?tqx=out:json`;

    async function loadMembers() {
        const response = await fetch(url);
        const text = await response.text();
        const json = JSON.parse(text.substr(47).slice(0, -2));
        const rows = json.table.rows;

        const container = document.getElementById('members-list');
        
        rows.forEach(row => {
            const name = row.c[0] ? row.c[0].v : '';
            const rank = row.c[1] ? row.c[1].v : '';
            const bio = row.c[2] ? row.c[2].v : '';

            const card = document.createElement('div');
            card.className = 'member-card';
            card.innerHTML = `
                <img src="https://api.dicebear.com/7.x/avataaars/svg?seed=${name}" alt="avatar">
                <h3>${name}</h3>
                <p>${rank}</p>
            `;
            card.onclick = () => showInfo(name, rank, bio);
            container.appendChild(card);
        });
    }

    function showInfo(name, rank, bio) {
        document.getElementById('modal-name').innerText = name;
        document.getElementById('modal-rank').innerText = rank;
        document.getElementById('modal-bio').innerText = bio;
        document.getElementById('info-modal').style.display = 'block';
        document.getElementById('overlay').style.display = 'block';
    }

    function closeModal() {
        document.getElementById('info-modal').style.display = 'none';
        document.getElementById('overlay').style.display = 'none';
    }

    loadMembers();
</script>
